36  	public JXBusyLabelTest() {
37  		bLabel1 = new JXBusyLabel(new Dimension(60, 60));
38  		bLabel1.setBusy(true);
39  
40  		String[] colorsName = {"Choose a color", "BLACK", "WHITE", "RED", "BLUE", "GREEN", "GRAY", "YELLOW"};
41  		comboBackground = new JComboBox(colorsName);
42  		comboForeground = new JComboBox(colorsName);
43  		btnStart = new JButton("Start");
44  		btnStop = new JButton("Stop");
45  
46  		comboBackground.addActionListener(this);
47  		comboForeground.addActionListener(this);
48  		btnStart.addActionListener(this);
49  		btnStop.addActionListener(this);
50  
51  		JPanel panelNorth = new JPanel();
52  		JPanel panelSouth = new JPanel();
53  		panelSouth.add(new JLabel("Background"));
54  		panelSouth.add(comboBackground);
55  		panelSouth.add(new JLabel("Foreground"));
56  		panelSouth.add(comboForeground);
57  		panelNorth.add(btnStart);
58  		panelNorth.add(btnStop);
59  
60  		add(panelNorth, BorderLayout.NORTH);
61  		add(bLabel1, BorderLayout.CENTER);
62  		add(panelSouth, BorderLayout.SOUTH);
63  
64  		setSize(400, 160);
65  		setDefaultCloseOperation(EXIT_ON_CLOSE);
66  		setVisible(true);
67  	}
68  
69  	public void actionPerformed(ActionEvent e) {
70  		if (e.getSource() == comboBackground) {
71  			bLabel1.getBusyPainter().setBaseColor(colors[comboBackground.getSelectedIndex()-1]);
72  		} else if (e.getSource() == comboForeground) {
73  			bLabel1.getBusyPainter().setHighlightColor(colors[comboForeground.getSelectedIndex()-1]);
74  		} else if (e.getSource() == btnStart) {
75  			bLabel1.setBusy(true);
76  		} else if (e.getSource() == btnStop) {
77  			bLabel1.setBusy(false);
78  		}
79  	}
